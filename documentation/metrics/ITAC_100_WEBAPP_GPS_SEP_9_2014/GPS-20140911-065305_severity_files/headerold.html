580     {

582       var oX, oY;

590       y = y * 180 / Math.PI;

596     function distance(point1, point2)
597      {
598       var R = 6371; // earth's mean radius in km
599       var lon1 = point1.lng()* Math.PI / 180;
600       var lat1 = point1.lat() * Math.PI / 180;
601       var lon2 = point2.lng() * Math.PI / 180;
602       var lat2 = point2.lat() * Math.PI / 180;
603       var deltaLat = lat1 - lat2;
604       var deltaLon = lon1 - lon2;
605       var step1 = Math.pow(Math.sin(deltaLat/2), 2) + Math.cos(lat2) * Math.cos(lat1) * Math.pow(Math.sin(deltaLon/2), 2);
606       var step2 = 2 * Math.atan2(Math.sqrt(step1), Math.sqrt(1 - step1));
607       return step2 * R;
