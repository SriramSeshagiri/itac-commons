527    /**
528     * <p>
529     * Encodes up to three bytes of the array <var>source</var> and writes the
530     * resulting four Base64 bytes to <var>destination</var>. The source and
531     * destination arrays can be manipulated anywhere along their length by
532     * specifying <var>srcOffset</var> and <var>destOffset</var>. This method
533     * does not check to make sure your arrays are large enough to accomodate
534     * <var>srcOffset</var> + 3 for the <var>source</var> array or
535     * <var>destOffset</var> + 4 for the <var>destination</var> array. The
536     * actual number of significant bytes in your array is given by
537     * <var>numSigBytes</var>.
538     * </p>
539     * <p>
540     * This is the lowest level of the encoding methods with all possible
541     * parameters.
542     * </p>
543     * 
544     * @param source
545     *            the array to convert
546     * @param srcOffset
547     *            the index where conversion begins
548     * @param numSigBytes
549     *            the number of significant bytes in your array
550     * @param destination
551     *            the array to hold the conversion
552     * @param destOffset
553     *            the index where output will be put
554     * @return the <var>destination</var> array
555     * @since 1.3
556     */
557    private static byte[] encode3to4(byte[] source, int srcOffset,
558          int numSigBytes, byte[] destination, int destOffset, int options) {
559
560       byte[] ALPHABET = getAlphabet(options);
561
562       // 1 2 3
563       // 01234567890123456789012345678901 Bit position
564       // --------000000001111111122222222 Array position from threeBytes
565       // --------| || || || | Six bit groups to index ALPHABET
566       // >>18 >>12 >> 6 >> 0 Right shift necessary
567       // 0x3f 0x3f 0x3f Additional AND
568
569       // Create buffer with zero-padding if there are only one or two
570       // significant bytes passed in the array.
571       // We have to shift left 24 in order to flush out the 1's that appear
572       // when Java treats a value as negative that is cast from a byte to an
573       // int.
574       int inBuff = (numSigBytes > 0 ? ((source[srcOffset] << 24) >>> 8) : 0)
575             | (numSigBytes > 1 ? ((source[srcOffset + 1] << 24) >>> 16) : 0)
576             | (numSigBytes > 2 ? ((source[srcOffset + 2] << 24) >>> 24) : 0);
577
578       switch (numSigBytes) {
579       case 3:
580          destination[destOffset] = ALPHABET[(inBuff >>> 18)];
581          destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
582          destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
583          destination[destOffset + 3] = ALPHABET[(inBuff) & 0x3f];
584          return destination;
585
586       case 2:
587          destination[destOffset] = ALPHABET[(inBuff >>> 18)];
588          destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
589          destination[destOffset + 2] = ALPHABET[(inBuff >>> 6) & 0x3f];
590          destination[destOffset + 3] = EQUALS_SIGN;
591          return destination;
592
593       case 1:
594          destination[destOffset] = ALPHABET[(inBuff >>> 18)];
595          destination[destOffset + 1] = ALPHABET[(inBuff >>> 12) & 0x3f];
596          destination[destOffset + 2] = EQUALS_SIGN;
597          destination[destOffset + 3] = EQUALS_SIGN;
598          return destination;
599
600       default:
601          return destination;
602       } // end switch
603    } // end encode3to4
