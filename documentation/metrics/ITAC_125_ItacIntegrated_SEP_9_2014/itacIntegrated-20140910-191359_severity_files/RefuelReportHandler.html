514    private double distanceCalculator(double slat, double slong,
515          double destlat, double destlong) {
516       double earthRadius = 6371.0;
517       double dLat = Math.toRadians(destlat - slat);
518       double dLng = Math.toRadians(destlong - slong);
519       double a = (Math.sin(dLat / 2) * Math.sin(dLat / 2) + Math.cos(Math
520             .toRadians(slat))
521             * Math.cos(Math.toRadians(destlat))
522             * Math.sin(dLng / 2) * Math.sin(dLng / 2));
523       double c = (2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a)));
524       double dist = (earthRadius * c);
525       return dist;
