672    /** Calculate distance in km between two coordinates **/
673    public static double distFrom(double lat1, double lng1, double lat2,
674          double lng2) {
675       double earthRadius = 3958.75;
676       double dLat = Math.toRadians(lat2 - lat1);
677       double dLng = Math.toRadians(lng2 - lng1);
678       double a = Math.sin(dLat / 2) * Math.sin(dLat / 2)
679             + Math.cos(Math.toRadians(lat1))
680             * Math.cos(Math.toRadians(lat2)) * Math.sin(dLng / 2)
681             * Math.sin(dLng / 2);
682       double c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
683       double dist = earthRadius * c;
684       double kmConversion = 1.609; // KMConversion constant to convert miles
685                               // to KM
686       double distance = (double) dist * kmConversion;
687
688       if (distance < 0.01)
689          return 0;
690       return distance;
691    }
