285    /**
286     * Computes the distance between two points.
287     * Returns the distance in km
288     * 
289     * @return Double 
290     */
291    public Double getDistance(double point1lat,double point1lon,double point2lat,double point2lon)
292    {
293       log.info("long: "+point1lon+ "lat: "+point1lat);
294       log.info("long: "+point2lon+ "lat: "+point2lat);
295      double R = 6371; // earth's mean radius in km
296       double lon1 = point1lon * Math.PI / 180;
297       double lat1 = point1lat * Math.PI / 180;
298       double lon2 = point2lon * Math.PI / 180;
299       double lat2 = point2lat * Math.PI / 180;
300       double deltaLat = lat1 - lat2;
301       double deltaLon = lon1 - lon2;
302       double step1 = Math.pow(Math.sin(deltaLat/2), 2) + Math.cos(lat2) * Math.cos(lat1) * Math.pow(Math.sin(deltaLon/2), 2);
303       double step2 = 2 * Math.atan2(Math.sqrt(step1), Math.sqrt(1 - step1));
304       return step2 * R;
305    }
