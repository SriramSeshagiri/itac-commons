282    /**
283     * Computes the distance between two points.
284     * Returns the distance in km
285     * 
286     * @return Double 
287     */
288    public Double getDistance(double point1lat,double point1lon,double point2lat,double point2lon)
289    {
290       log.info("long: "+point1lon+ "lat: "+point1lat);
291       log.info("long: "+point2lon+ "lat: "+point2lat);
292      double R = 6371; // earth's mean radius in km
293       double lon1 = point1lon * Math.PI / 180;
294       double lat1 = point1lat * Math.PI / 180;
295       double lon2 = point2lon * Math.PI / 180;
296       double lat2 = point2lat * Math.PI / 180;
297       double deltaLat = lat1 - lat2;
298       double deltaLon = lon1 - lon2;
299       double step1 = Math.pow(Math.sin(deltaLat/2), 2) + Math.cos(lat2) * Math.cos(lat1) * Math.pow(Math.sin(deltaLon/2), 2);
300       double step2 = 2 * Math.atan2(Math.sqrt(step1), Math.sqrt(1 - step1));
301       return step2 * R;
302    }
