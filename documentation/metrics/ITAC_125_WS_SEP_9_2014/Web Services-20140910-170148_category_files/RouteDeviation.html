311    private boolean isBkOnRefRoutePoint(int startIndex, int refTripPointId,
312          TripReference currentPacketPoint) throws Exception {
313
314       boolean backOnRefRouteFlag = false;
315
316       @SuppressWarnings("unchecked")
317       Collection<TripReference> cRoutePointsInSequence = referenceRoute
318             .values();
319       Iterator<TripReference> itr = cRoutePointsInSequence.iterator();
320
321       for (int i = 0; (itr.hasNext()); i++) {
322          TripReference referenceRoutePoint = (TripReference) (itr.next());
323          if (i >= startIndex) {
324
325             // point has neither been traversed nor deviated so check
326             // whether it is in the permissible distance
327             boolean bPointInCircle = isPointInCircle(currentPacketPoint,
328                   referenceRoutePoint);
329             
330             if (bPointInCircle) {
331
332                Calendar cal1 = Calendar.getInstance();
333                int timeDiffInSecs = 0;
334                // Get the difference between traversed packet time and
335                // corresponding reference packet time
336                int realTimeDiff = ((new TraversalDAO())
337                      .getTimeDelta(currentPacketPoint));
338
339                int timeDelta = dHelper.getPermissibleTimeDelta();
340                if (startIndex == 0)
341                   timeDelta = 1800;
342
343                // Get the permissible time delta
344                // int timeDelta = dHelper.getPermissibleTimeDelta();
345
346                // As per the real time difference between traversed and
347                // reference points need to adjust the reference point time
348                // before
349                // comparing with current packet time. Hence add the real
350                // time difference to reference point time
351                cal1.setTime(referenceRoutePoint.getTripPointTime());
352                int refPtTimeInSecs = ((cal1.get(Calendar.HOUR_OF_DAY)) * 60 * 60)
353                      + (cal1.get(Calendar.MINUTE))
354                      * 60
355                      + (cal1.get(Calendar.SECOND)) + realTimeDiff;
356
357                // Get the current packet time in seconds
358                cal1.setTime(currentPacketPoint.getTripPointTime());
359                int currPktPtTimeInSecs = ((cal1.get(Calendar.HOUR_OF_DAY)) * 60 * 60)
360                      + (cal1.get(Calendar.MINUTE))
361                      * 60
362                      + (cal1.get(Calendar.SECOND));
363
364                // If current packet time is less than reference point time
365                // then subtract it from reference point time to get the
366                // difference otherwise vice-versa
367                if (currPktPtTimeInSecs >= refPtTimeInSecs)
368                   timeDiffInSecs = currPktPtTimeInSecs - refPtTimeInSecs;
369                else
370                   timeDiffInSecs = refPtTimeInSecs - currPktPtTimeInSecs;
371
372                // if difference is greater than permissible time delta then
373                // check for recurrent locations
374                int trpMatchPointId = referenceRoutePoint.getTripPointId();
375                if (timeDiffInSecs > timeDelta) {
376
377                   if ((referenceRoutePoint.getTripRecurrentLocations())
378                         .length() > 0) {
379                      // if there are recurrent locations then find out
380                      // which is closest to current point
381                      trpMatchPointId = getClosestPointOnRefRoute(
382                            referenceRoutePoint, currentPacketPoint);
383
384                      // need to check that the matched point is not
385                      // already traversed
386                      boolean bTraversed = (new TraversalDAO())
387                            .isPointAlreadyTraversed(currentPacketPoint
388                                  .getTripDetails().getTripid(),
389                                  trpMatchPointId, currentPacketPoint
390                                        .getTripSession(),
391                                  currentPacketPoint
392                                        .getTripPointTime());
393
394                      // if traversed then it is a route deviation hence
395                      // break loop and return deviation true
396                      if (bTraversed) {
397                         backOnRefRouteFlag = false;
398                         break;
399                      }
400                   }
401
402                   // Suppose if the matched point is same as the earlier
403                   // matched reference point then as the time difference
404                   // was more it will anyways continue checking against
405                   // next point
406                   if (referenceRoutePoint.getTripPointId() != trpMatchPointId) {
407                      // if the matched point is different then we need to
408                      // check its time against current packet
409                      // again here adjust matched point time by adding
410                      // real time difference.
411                      cal1.setTime((new TripReferenceDAO()).findById(
412                            trpMatchPointId).getTripPointTime());
413                      int matchPtTimeInSecs = ((cal1
414                            .get(Calendar.HOUR_OF_DAY)) * 60 * 60)
415                            + (cal1.get(Calendar.MINUTE))
416                            * 60
417                            + (cal1.get(Calendar.SECOND))
418                            + realTimeDiff;
419                      if (currPktPtTimeInSecs >= matchPtTimeInSecs)
420                         timeDiffInSecs = currPktPtTimeInSecs
421                               - matchPtTimeInSecs;
422                      else
423                         timeDiffInSecs = matchPtTimeInSecs
424                               - currPktPtTimeInSecs;
425
426                   }
427                }
428                // Now if time difference between two points is less than
429                // permissible time delta then mark matched point as
430                // traversed
431                // and all the points from the reference point to matched
432                // point as deviated approved.
433                if (timeDiffInSecs <= timeDelta) {
434                   TripTraversalDAO trpTraversalDAO = new TripTraversalDAO();
435                   Session sess = trpTraversalDAO.getSession();
436
437                   Transaction tx = sess.beginTransaction();
438                   tx.begin();
439                   int numberOfPointsTraversed = startIndex;
440                   for (int pointId = refTripPointId; (pointId < trpMatchPointId && (numberOfPointsTraversed < (referenceRoute
441                         .size() - 1))); pointId++) {
442                      TripTraversal objTripTraversal = new TripTraversal();
443                      TripReference tripReference = new TripReference();
444                      tripReference.setTripPointId(pointId);
445
446                      objTripTraversal.setTripDetails(currentPacketPoint
447                            .getTripDetails());
448                      objTripTraversal.setTripReference(tripReference);
449                      objTripTraversal.setHasTraversed(false);
450                      objTripTraversal.setIsSupportTeamOk(1);
451                      objTripTraversal
452                            .setTripPointTraversalTime(currentPacketPoint
453                                  .getTripPointTime());
454
455                      sess.save(objTripTraversal);
456                      numberOfPointsTraversed++;
457                   }
458
459                   // Last entry in trip traversal table will be the
460                   // reference route point
461                   // that matches with current packet is to be marked as
462                   // traversed.
463                   TripTraversal objTripTraversal = new TripTraversal();
464                   TripReference tripReference = new TripReference();
465                   tripReference.setTripPointId(trpMatchPointId);
466
467                   objTripTraversal.setTripDetails(currentPacketPoint
468                         .getTripDetails());
469                   objTripTraversal.setTripReference(tripReference);
470                   objTripTraversal.setHasTraversed(true);
471                   objTripTraversal.setIsSupportTeamOk(0);
472                   objTripTraversal
473                         .setTripPointTraversalTime(currentPacketPoint
474                               .getTripPointTime());
475
476                   sess.save(objTripTraversal);
477
478                   // Make an entry in trip deviation as the deviated path
479                   // was shorter than reference path
480                   TripDeviation objTripDeviation = new TripDeviation();
481                   objTripDeviation.setIsSupportTeamOk(5);
482                   objTripDeviation.setTripDetails(currentPacketPoint
483                         .getTripDetails());
484                   objTripDeviation.setTime(currentPacketPoint
485                         .getTripPointTime());
486
487                   sess.save(objTripDeviation);
488
489                   tx.commit();
490                   sess.close();
491
492                   backOnRefRouteFlag = true;
493                   break;
494                }
495             } else {
496                continue;
497             }
498          }
499       }
500       return backOnRefRouteFlag;
501    }

610     * @throws Exception
